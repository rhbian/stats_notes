\section{Beta分布}

\begin{definition}{The Beta Function}{}
for each positive $\alpha$ and $\beta$, define
\begin{equation}\label{}
B(\alpha, \beta)=\int_{0}^{1} x^{\alpha-1}(1-x)^{\beta-1}dx
\end{equation}
the function $B$ is called the \textit{beta function}.

\end{definition}
